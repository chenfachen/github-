## 快速排序


## 代码
```java
class Solution {
    public int[] sortArray(int[] nums) {
        int l = 0;
        int r = nums.length - 1;
        int pivot = partition(nums, l, r);
        return nums;
    }

    public int partition(int[] nums, int l, int r){
        if(l >= r){
            return l;
        }
        int pivot = nums[l];
        int i = l;
        int j = r + 1;
        while(true){
            while(nums[++i] < pivot){
                if(i == r) break;
            }
            while(nums[--j] > pivot){
                if(j == l) break;
            }
            if(i >= j) break;
            swap(nums, i, j);
        }

        swap(nums, l, j);
        partition(nums, l, j-1);
        partition(nums, j+1, r);
        return j;                 //可以没有返回值
    }

    public void swap(int[] nums, int i, int j){
        int tmp = nums[i];
        nums[i] = nums[j];
        nums[j] = tmp;
    }
}
```
